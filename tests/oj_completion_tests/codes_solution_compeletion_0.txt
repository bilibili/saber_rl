def main():
    import sys
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    n = int(input_data[0])
    w = int(input_data[1])
    cups = list(map(int, input_data[2:]))
    cups.sort()

    min_boy = cups[n]
    
    x_possible = min(min_girl, min_boy / 2, w / (3 * n))
    total_water = 3 * n * x_possible

if __name__ == "__main__":
