from solution import max_product

def test_max_product_positive_numbers():
    assert max_product([1, 5, 3, 7, 9]) == 63  # 7 * 9 = 63

def test_max_product_with_negatives():
    assert max_product([-1, -3, -5, 2, 4]) == 15  # -3 * -5 = 15

def test_max_product_mixed_signs():
    assert max_product([5, 3, -1, 9, -7]) == 45  # 5 * 9 = 45

def test_max_product_two_numbers():
    assert max_product([2, 3]) == 6  # 2 * 3 = 6

def test_max_product_single_zero():
    assert max_product([0, 5, 10]) == 50  # 5 * 10 = 50

def test_max_product_zeroes():
    assert max_product([0, 0, 2, 3]) == 6  # 2 * 3 = 6

def test_max_product_large_numbers():
    assert max_product([1000, 3000, 5000, 2000]) == 15000000  # 3000 * 5000 = 15000000

def test_max_product_identical_elements():
    assert max_product([5, 5, 5, 5]) == 25  # 5 * 5 = 25
